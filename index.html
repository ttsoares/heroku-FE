Operação Descrição URL
GET Mostra Um usuário ('/users/:id')
GET Mostra todos os usuários ('/users')
GET Saldo de um usuário ('/user/:id')
GET Uma transação de um usuário ('/user/:userid/transaction/:transid)
POST Cadastra um usuário ('/users')
POST Cadastra uma transação ('/user/:userid/transactions)
PUT Modifica usuário ('/user/:id')
PUT Modifica transação ('/user/:userid/transaction/:transid)
DELETE Apaga um usuário ('/user/:id')
DELETE Apaga uma transação ('/user/:userid/transaction/:transid)
